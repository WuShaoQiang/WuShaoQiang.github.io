排序时间复杂度

堆排序

参考dreamcatcher-cx博客

最好最坏的平均时间复杂度都为O(NlogN)

代码

func HeapSortUp(a []int) []int {
	for i := len(a)/2 - 1; i >= 0; i-- {
		adjustHeapUp(a, i, len(a))
	}

	for j := len(a) - 1; j > 0; j-- {
		a[0], a[j] = a[j], a[0]
		adjustHeapUp(a, 0, j)
	}
	return a
}

func adjustHeapUp(a []int, i int, length int) {
	tmp := a[i]
	for k := 2*i + 1; k < length; k = 2*k + 1 {
		if k+1 < length && a[k+1] > a[k] {
			k++
		}

		if a[k] > tmp {
			a[i] = a[k]
			i = k
		} else {
			break
		}
	}
	a[i] = tmp
}

思路

  1. 先将数组变换为一个大顶堆

  2. 将大顶堆的顶部和堆的最后一个元素交换

  3. 再将堆变成大顶堆，重复2，直到最后一个元素

注意

初次构造大顶堆和重构大顶堆不一样

  • 初次构造大顶堆需要从下到上，从右到左

  • 重构大顶堆从0开始

package chapter7

func MergeSort(a []int) []int {
	tmp := make([]int, len(a))
	sort(a, 0, len(a)-1, tmp)
	return a

}

func sort(a []int, left, right int, tmp []int) {
	if left < right {
		mid := (right + left) / 2
		sort(a, left, mid, tmp)
		sort(a, mid+1, right, tmp)
		merge(a, left, mid, right, tmp)
	}
}

func merge(a []int, left, mid, right int, tmp []int) {
	l := left
	r := mid + 1
	t := 0

	for l <= mid && r <= right {
		if a[l] <= a[r] {
			tmp[t] = a[l]
			t++
			l++
		} else {
			tmp[t] = a[r]
			t++
			r++
		}
	}

	for l <= mid {
		tmp[t] = a[l]
		t++
		l++
	}
	for r <= right {
		tmp[t] = a[r]
		t++
		r++
	}

	t = 0
	for left <= right {
		a[left] = tmp[t]
		left++
		t++
	}
}

思路

  1. 递归将其拆成一个元素，通过merge排序两个元素

  2. 将排序好的替换到原数组上

  3. 现在每两个元素之间是有序的了，继续将两个部分进行merge排序，循环2…

希尔排序

代码

func ShellSort(a []int) []int {
	for increment := len(a) / 2; increment > 0; increment = increment / 2 {
		for i := increment; i < len(a); i++ {
			//插入排序
			temp := a[i]
			var j int
			for j = i; j >= increment && a[j-increment] > temp; j -= increment {
				a[j] = a[j-increment]
			}
			a[j] = temp
		}
	}
	return a
}

思路

  1. 首先希尔排序每次gap是之前的1/2

  2. 从gap索引开始，将后面的元素按照gap的空隙进行直接插入排序

快速排序

代码

func QuickSort(a []int) []int {
	quickSort(a, 0, len(a)-1)
	return a
}

func quickSort(a []int, left, right int) {
	if left < right {
		low := left
		high := right
		tmp := a[low]
		var leftSide = false

		for low < high {
			if !leftSide {
				if a[high] < tmp {
					a[low] = a[high]
					low++
					leftSide = true
				} else {
					high--
				}
			} else {
				if a[low] > tmp {
					a[high] = a[low]
					high--
					leftSide = false
				} else {
					low++
				}
			}
		}
		a[low] = tmp

		quickSort(a, left, low-1)
		quickSort(a, low+1, right)
	}
}

func quickSort(sortArray []int, left, right int) {

	if left < right {
		key := sortArray[left]
		i := left
		j := right

		for {
			for i+1 < MAX {
				i++
				if key <= sortArray[i] {
					break
				}
			}

			for j-1 >= 0 {
				if key >= sortArray[j] {
					break
				}
				j--
			}

			if i >= j {
				break
			}

			swap(i, j)
		}

		sortArray[left] = sortArray[j]
		sortArray[j] = key
		// show()

		quickSort(sortArray, left, j-1)
		quickSort(sortArray, j+1, right)
	}

}

改进：之前是左边一次右边一次的执行交换，改进后是取得两边的值后再交换，最后将key的值放到和比key小的最后一个数进行交换(类似放在中央，左边是比它小的数，右边是比它大的数)

思路

  1. 取出第一个数作为参照

  2. 创建左右索引，将小于参照的放左边，大于参照的放右边

func InsertSortUp(a []int) []int {
	var j int
	for i := 1; i < len(a); i++ {
		tmp := a[i]
		for j = i; j > 0 && tmp < a[j-1]; j-- {
			a[j] = a[j-1]
		}
		a[j] = tmp
	}
	return a
}

思路

  1. 插入排序在找到比这个数还小的前面进行插入
