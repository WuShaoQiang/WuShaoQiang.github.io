趁着周末有空，抽几个小时看看源码，这次看的是raftexample

这是一个raft的简单实现，etcd/raft只是实现了raft的主要算法(例如选举、状态机等)

To keep the codebase small as well as provide flexibility, the library only implements the Raft algorithm; both network and disk IO are left to the user. Library users must implement their own transportation layer for message passing between Raft peers over the wire. Similarly, users must implement their own storage layer to persist the Raft log and state.

因此我觉得先看一个完整的实例可能对raft的整个逻辑会有一个清晰的认识

因此我们可以先从怎么用etcd的raft来入手

这篇blog的范围仅在raftexample这部分，超过这部分的代码(例如raft和node这些细节，在后面的blog会慢慢接触)暂时不会说

我们继续追踪下去

func (rc *raftNode) startRaft() {
    // snapdir是存储Snapshot的地方
	if !fileutil.Exist(rc.snapdir) {
		if err := os.Mkdir(rc.snapdir, 0750); err != nil {
			log.Fatalf("raftexample: cannot create dir for snapshot (%v)", err)
		}
	}
    // snapshotter是一个读写Snapshot的结构体，有Load和Save
	rc.snapshotter = snap.New(zap.NewExample(), rc.snapdir)
	rc.snapshotterReady <- rc.snapshotter

    // WAL(Write Ahead Log)是一种先记录log再写入数据的一种方法
    // 这里是为了决定是Restart旧的还是Start新的
	oldwal := wal.Exist(rc.waldir)
    // 这里会创建或者恢复一个WAL
	rc.wal = rc.replayWAL()

	rpeers := make([]raft.Peer, len(rc.peers))
	for i := range rpeers {
		rpeers[i] = raft.Peer{ID: uint64(i + 1)}
	}
	c := &raft.Config{
		// 一些赋值操作...
	}

    // 如上面所说，会因为WAL是否存在而走不同的启动逻辑
	if oldwal {
		rc.node = raft.RestartNode(c)
	} else {
		startPeers := rpeers
		if rc.join {
			startPeers = nil
		}
		rc.node = raft.StartNode(c, startPeers)
	}

	// 一些transport的配置

    // 主要逻辑
	go rc.serveRaft()
	go rc.serveChannels()
}

来看一下我们应该注意哪些

  • 首先snapshotter和wal先不用关心，只需要知道它们的作用就好

  • Restart逻辑和Start逻辑是很相近的，我们可以看作一个

  • 下面两个函数是主要逻辑，需要着重关注

无论是Restart还是Start，最终返回的都是一个Node的接口，这个接口定义了一个raft cluster上每一个node的行为，所以我们只需要知道，无论是Restart还是Start，都是整个Raft Cluster的一个节点

这也验证了一个main.go代表的只是一个node的启动以及行为

来到主要逻辑部分

首先是serveRaft，一般带serve的很容易和服务联系起来，这里就是将一个node的监听端口打开，启动服务

func (rc *raftNode) serveRaft() {
    // 解析自己的url
	url, err := url.Parse(rc.peers[rc.id-1])
	if err != nil {
		log.Fatalf("raftexample: Failed parsing URL (%v)", err)
	}

    // stoppableListener是一个带有close chan的Listener
	ln, err := newStoppableListener(url.Host, rc.httpstopc)
	if err != nil {
		log.Fatalf("raftexample: Failed to listen rafthttp (%v)", err)
	}

    // 启动服务
	err = (&http.Server{Handler: rc.transport.Handler()}).Serve(ln)
	select {
	case <-rc.httpstopc:
	default:
		log.Fatalf("raftexample: Failed to serve rafthttp (%v)", err)
	}
	close(rc.httpdonec)
}

在这里我们可以大概看一下这个stoppableListener是如何处理连接的

func (ln stoppableListener) Accept() (c net.Conn, err error) {
	connc := make(chan *net.TCPConn, 1)
	errc := make(chan error, 1)
	go func() {
		tc, err := ln.AcceptTCP()
		if err != nil {
			errc <- err
			return
		}
		connc <- tc
	}()
	select {
        // 实现从外部接收信号来关闭服务
	case <-ln.stopc:
		return nil, errors.New("server stopped")
        // 发生错误也会关闭服务
	case err := <-errc:
		return nil, err
        // 正常情况下就是返回一个连接
	case tc := <-connc:
		tc.SetKeepAlive(true)
		tc.SetKeepAlivePeriod(3 * time.Minute)
		return tc, nil
	}
}

可以看到，只是在原来基础上的Accept加多了两种的处理逻辑，其正常的逻辑和未被覆盖的Accept是一样的

我们总结起来，这个函数是为了打开服务，具体的Handler先不理会

再来看一下serveChannels这个函数，这个函数才是业务的主要处理逻辑

func (rc *raftNode) serveChannels() {
    // raftStorage在replayWAL的时候被初始化了
	snap, err := rc.raftStorage.Snapshot()
	if err != nil {
		panic(err)
	}
    // 节点的状态赋值
	rc.confState = snap.Metadata.ConfState
	rc.snapshotIndex = snap.Metadata.Index
	rc.appliedIndex = snap.Metadata.Index

	defer rc.wal.Close()

	ticker := time.NewTicker(100 * time.Millisecond)
	defer ticker.Stop()

	// send proposals over raft
	go func() {
		confChangeCount := uint64(0)

        // 接收来自Propose Chan(包括Msg和ConfChange)来的值
        // 并且进入对应的逻辑
		for rc.proposeC != nil && rc.confChangeC != nil {
			select {
			case prop, ok := <-rc.proposeC:
				if !ok {
					rc.proposeC = nil
				} else {
					// blocks until accepted by raft state machine
					rc.node.Propose(context.TODO(), []byte(prop))
				}

			case cc, ok := <-rc.confChangeC:
				if !ok {
					rc.confChangeC = nil
				} else {
					confChangeCount++
					cc.ID = confChangeCount
					rc.node.ProposeConfChange(context.TODO(), cc)
				}
			}
		}
		// client closed channel; shutdown raft if not already
		close(rc.stopc)
	}()

	// event loop on raft state machine updates
    // 主要是State machine的处理
	for {
		select {
            // 定时的Tick，用来记录时间
		case <-ticker.C:
			rc.node.Tick()

		// store raft entries to wal, then publish over commit channel
		case rd := <-rc.node.Ready():
			rc.wal.Save(rd.HardState, rd.Entries)
			if !raft.IsEmptySnap(rd.Snapshot) {
				rc.saveSnap(rd.Snapshot)
				rc.raftStorage.ApplySnapshot(rd.Snapshot)
				rc.publishSnapshot(rd.Snapshot)
			}
			rc.raftStorage.Append(rd.Entries)
			rc.transport.Send(rd.Messages)
			if ok := rc.publishEntries(rc.entriesToApply(rd.CommittedEntries)); !ok {
				rc.stop()
				return
			}
			rc.maybeTriggerSnapshot()
			rc.node.Advance()

		case err := <-rc.transport.ErrorC:
			rc.writeError(err)
			return

		case <-rc.stopc:
			rc.stop()
			return
		}
	}
}
