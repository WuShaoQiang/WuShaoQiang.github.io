给定一个字符串，找出其中不含有重复字符的最长子串的长度方法 - 穷举法：

func lengthOfLongestSubstring(s string) int {
	var max = 0
	for i := 0; i < len(s); i++ {
		m := make(map[string]bool)
		for j := i; j < len(s); j++ {
			if _, ok := m[s[j:j+1]]; !ok {
				m[s[j:j+1]] = true
				if j == len(s)-1 {
					if max < j-i+1 {
						max = j - i + 1
					}
				}
			} else {
				if max < j-i {
					max = j - i
				}
				break
			}
		}
	}
	return max
}
