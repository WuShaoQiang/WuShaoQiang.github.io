代码

func HeapSortUp(a []int) []int {
	for i := len(a)/2 - 1; i >= 0; i-- {
		adjustHeapUp(a, i, len(a))
	}

	for j := len(a) - 1; j > 0; j-- {
		a[0], a[j] = a[j], a[0]
		adjustHeapUp(a, 0, j)
	}
	return a
}

func adjustHeapUp(a []int, i int, length int) {
	tmp := a[i]
	for k := 2*i + 1; k < length; k = 2*k + 1 {
		if k+1 < length && a[k+1] > a[k] {
			k++
		}

		if a[k] > tmp {
			a[i] = a[k]
			i = k
		} else {
			break
		}
	}
	a[i] = tmp
}

思路

  1. 先将数组变换为一个大顶堆

  2. 将大顶堆的顶部和堆的最后一个元素交换

  3. 再将堆变成大顶堆，重复2，知道最后一个元素

注意

初次构造大顶堆和重构大顶堆不一样

  • 初次构造大顶堆需要从下到上，从右到左

  • 重构大顶堆从0开始

package chapter7

func MergeSort(a []int) []int {
	tmp := make([]int, len(a))
	sort(a, 0, len(a)-1, tmp)
	return a

}

func sort(a []int, left, right int, tmp []int) {
	if left < right {
		mid := (right + left) / 2
		sort(a, left, mid, tmp)
		sort(a, mid+1, right, tmp)
		merge(a, left, mid, right, tmp)
	}
}

func merge(a []int, left, mid, right int, tmp []int) {
	l := left
	r := mid + 1
	t := 0

	for l <= mid && r <= right {
		if a[l] <= a[r] {
			tmp[t] = a[l]
			t++
			l++
		} else {
			tmp[t] = a[r]
			t++
			r++
		}
	}

	for l <= mid {
		tmp[t] = a[l]
		t++
		l++
	}
	for r <= right {
		tmp[t] = a[r]
		t++
		r++
	}

	t = 0
	for left <= right {
		a[left] = tmp[t]
		left++
		t++
	}
}

思路

  1. 递归将其拆成一个元素，通过merge排序两个元素

  2. 将排序好的替换到原数组上

  3. 现在每两个元素之间是有序的了，继续将两个部分进行merge排序，循环2…

func ShellSort(a []int) []int {
	for increment := len(a) / 2; increment > 0; increment = increment / 2 {
		for i := increment; i < len(a); i++ {
			//插入排序
			temp := a[i]
			var j int
			for j = i; j >= increment && a[j-increment] > temp; j -= increment {
				a[j] = a[j-increment]
			}
			a[j] = temp
		}
	}
	return a
}
